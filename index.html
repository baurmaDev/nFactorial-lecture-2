    const LockersComponent = () => {
        const [lockers, setLockers] = useState([
            { id: 1, label: 'Bauyrzhan', status: 'closed' },
            { id: 2, label: 'Jarasar', status: 'open' },
            // другие шкафчики
        ]);
    
        const handleUpdate = (lockerId, newStatus) => {
            setLockers(prevLockers =>
                prevLockers.map(locker =>
                    locker.id === lockerId ?
                     { ...locker, status: newStatus }
                      : locker
                )
            );
        };
    
        return (
            <div>
                {lockers.map(locker => (
                    <Locker key={locker.id} locker={locker} onUpdate={handleUpdate} />
                ))}
            </div>
        );
    };
    
    export default LockersComponent;
